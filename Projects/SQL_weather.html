Global Weather Dashboard with SQL

This project visualises the current weather conditions across multiple cities worldwide using data stored in a SQL database. By querying the current_weather and cities tables, the relevant temperature and weather descriptions are extracted and displayed on an interactive map using Plotly. Marker colours represent temperature variations, and hovering over each city reveals detailed weather information. In addition to visualisation, SQL queries are used to generate insights such as identifying the hottest and coldest cities, calculating average temperatures by region, and summarising weather conditions across multiple locations. The project demonstrates how SQL can be combined with Python data visualisation tools to create dynamic, informative dashboards that are easily exportable to HTML for sharing or reporting purposes.

# function to get current weather
def get_current_weather(city):
    params = {
        "q": city,
        "appid": OPENWEATHER_API_KEY,
        "units": "metric"
    }
    response = requests.get("https://api.openweathermap.org/data/2.5/weather", params=params)
    return response.json()

# function to get forecast
def get_forecast(city):
    params = {
        "q": city,
        "appid": OPENWEATHER_API_KEY,
        "units": "metric"
    }
    response = requests.get("https://api.openweathermap.org/data/2.5/forecast", params=params)
    return response.json()

# Create tables
cursor.executescript("""
                     
CREATE TABLE IF NOT EXISTS cities (
    city_id INTEGER PRIMARY KEY,
    name TEXT,
    country TEXT,
    lat REAL,
    lon REAL
);

CREATE TABLE IF NOT EXISTS current_weather (
    weather_id INTEGER PRIMARY KEY AUTOINCREMENT,
    city_id INTEGER,
    date_time TEXT,
    temp REAL,
    humidity REAL,
    weather_desc TEXT,
    wind_speed REAL,
    FOREIGN KEY(city_id) REFERENCES cities(city_id)
);

CREATE TABLE IF NOT EXISTS forecast (
    forecast_id INTEGER PRIMARY KEY AUTOINCREMENT,
    city_id INTEGER,
    forecast_time TEXT,
    temp_min REAL,
    temp_max REAL,
    weather_desc TEXT,
    FOREIGN KEY(city_id) REFERENCES cities(city_id)
);
""")

conn.commit()

# Insert city data
def insert_city(data):
    cursor.execute("""
        INSERT OR IGNORE INTO cities (city_id, name, country, lat, lon)
        VALUES (?, ?, ?, ?, ?)
    """, (
        data["id"],
        data["name"],
        data["sys"]["country"],
        data["coord"]["lat"],
        data["coord"]["lon"]
    ))
    conn.commit()

# Insert current weather
def insert_current_weather(data):
    cursor.execute("""
        INSERT INTO current_weather (city_id, date_time, temp, humidity, weather_desc, wind_speed)
        VALUES (?, ?, ?, ?, ?, ?)
    """, (
        data["id"],
        datetime.fromtimestamp(data["dt"], tz=timezone.utc).isoformat(),  
        data["main"]["temp"],
        data["main"]["humidity"],
        data["weather"][0]["description"],
        data["wind"]["speed"]
    ))
    conn.commit()

# Insert forecast data (multiple entries)
def insert_forecast(data):
    for entry in data["list"]:
        cursor.execute("""
            INSERT INTO forecast (city_id, forecast_time, temp_min, temp_max, weather_desc)
            VALUES (?, ?, ?, ?, ?)
        """, (
            data["city"]["id"],
            datetime.fromtimestamp(entry["dt"], tz=timezone.utc).isoformat(),
            entry["main"]["temp_min"],
            entry["main"]["temp_max"],
            entry["weather"][0]["description"]
        ))
    conn.commit()

def SQL(sql_prompt, parse_dates=None):
    """
    Execute SQL and return a Pandas DataFrame.
    
    Parameters:
        sql_prompt (str): SQL query to execute.
        parse_dates (list or None): Column names to parse as datetime.
    
    Returns:
        pd.DataFrame: Query results.
    """
    return pd.read_sql(sql_prompt, conn, parse_dates=parse_dates)

Hottest_City_24h = """
    SELECT c.name, MAX(f.temp_max) AS max_temp
    FROM forecast f
    JOIN cities c USING(city_id)
    WHERE f.forecast_time >= datetime('now')
      AND f.forecast_time <= datetime('now', '+1 day')
    GROUP BY c.name
    ORDER BY max_temp DESC
    LIMIT 1;
    """

# Query: max temperature forecasts for plotting
query = """
SELECT c.name, f.forecast_time, f.temp_max
FROM forecast f
JOIN cities c USING(city_id)
ORDER BY c.name, f.forecast_time
"""

# Get DataFrame and parse datetime
df = SQL(query, parse_dates=['forecast_time'])

# Find top 3 cities by average max temp
top_cities = df.groupby('name')['temp_max'].mean().nlargest(3).index.tolist()

# Plot only top 3 cities
plt.figure(figsize=(10,6))
for city in top_cities:
    city_data = df[df['name'] == city]
    plt.plot(city_data['forecast_time'], city_data['temp_max'], label=city)

plt.xlabel("Forecast Time")
plt.ylabel("Max Temperature (C)")
plt.title("5-Day Max Temperature Forecast - Top 3 Hottest Cities")
plt.legend()
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()

# SQL query: take first forecast entry per city for simplicity
query_map = """
SELECT c.name, c.lat, c.lon, f.temp_min, f.temp_max, f.weather_desc
FROM cities c
JOIN forecast f 
  ON c.city_id = f.city_id
WHERE f.forecast_time = (
    SELECT MIN(f2.forecast_time)
    FROM forecast f2
    WHERE f2.city_id = c.city_id
)
"""
df_map = SQL(query_map)

fig = px.scatter_geo(
    df_map,
    lat='lat',
    lon='lon',
    color='temp_max',
    size='temp_max',
    hover_name='name',
    hover_data={'temp_min': True, 'temp_max': True, 'weather_desc': True, 'lat': False, 'lon': False},
    projection='natural earth',
    color_continuous_scale="Viridis",
    size_max=20,  
    opacity=0.8
)

fig.update_layout(
    geo=dict(
        showland=True,
        landcolor="lightgray",
        showocean=True,
        oceancolor="lightblue",
        showcountries=True,
        countrycolor="white"
    )
)
